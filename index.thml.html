<h1>找因數遊戲</h1>

<div id="target-number">請找出數字 <strong id="number"></strong> 的所有因數</div>

<script>

    const numberElement = document.getElementById('number');

    const optionsContainer = document.getElementById('options');

    const correctCountElement = document.getElementById('correct-count');

    const incorrectCountElement = document.getElementById('incorrect-count');

    const nextButton = document.getElementById('next-button');


    let targetNumber;

    let factors = [];

    let nonFactors = [];

    let options = [];

    let correctSelections = 0;

    let incorrectSelections = 0;


    function startGame() {

        resetGame();

        generateQuestion();

        displayOptions();

    }


    function resetGame() {

        optionsContainer.innerHTML = '';

        nextButton.style.display = 'none';

        correctSelections = 0;

        incorrectSelections = 0;

        correctCountElement.textContent = correctSelections;

        incorrectCountElement.textContent = incorrectSelections;

    }


    function generateQuestion() {

        // 可以修改這裡的範圍，生成不同的目標數字

        targetNumber = Math.floor(Math.random() * 90) + 10; // 10 到 99 的數字

        numberElement.textContent = targetNumber;


        factors = getFactors(targetNumber);

        nonFactors = generateNonFactors(targetNumber, factors);


        options = factors.concat(nonFactors);

        shuffle(options);

    }


    // 獲取一個數字的所有因數

    function getFactors(num) {

        let factors = [];

        for (let i = 1; i <= Math.floor(num / 2); i++) {

            if (num % i === 0) {

                factors.push(i);

            }

        }

        factors.push(num); // 加上自己本身

        return factors;

    }


    // 生成非因數的干擾項

    function generateNonFactors(num, factors) {

        let nonFactors = [];

        while (nonFactors.length < factors.length) {

            let randomNum = Math.floor(Math.random() * 90) + 2; // 2 到 99 的數字

            if (num % randomNum !== 0 && !nonFactors.includes(randomNum)) {

                nonFactors.push(randomNum);

            }

        }

        return nonFactors;

    }


    // 隨機排列數組

    function shuffle(array) {

        array.sort(() => Math.random() - 0.5);

    }


    function displayOptions() {

        options.forEach(value => {

            const optionDiv = document.createElement('div');

            optionDiv.classList.add('option');

            optionDiv.textContent = value;

            optionDiv.addEventListener('click', () => selectOption(optionDiv, value));

            optionsContainer.appendChild(optionDiv);

        });

    }


    function selectOption(element, value) {

        if (element.classList.contains('correct') || element.classList.contains('incorrect')) {

            return;

        }


        if (factors.includes(value)) {

            element.classList.add('correct');

            playSound('correct');

            correctSelections++;

            correctCountElement.textContent = correctSelections;

            checkCompletion();

        } else {

            element.classList.add('incorrect');

            playSound('wrong');

            incorrectSelections++;

            incorrectCountElement.textContent = incorrectSelections;

        }

    }


    function checkCompletion() {

        const allCorrectSelected = document.querySelectorAll('.correct').length === factors.length;

        if (allCorrectSelected) {

            nextButton.style.display = 'inline-block';

        }

    }


    nextButton.addEventListener('click', () => {

        resetGame();

        generateQuestion();

        displayOptions();

    });


    // 播放音效

    function playSound(type) {

        const sound = document.getElementById(type + '-sound');

        sound.currentTime = 0;

        sound.play();

    }


    // 開始遊戲

    startGame();
